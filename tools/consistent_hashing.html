1. The ring
Hash space 0–2³² mapped to a circle. Nodes and keys both hashed onto it using MD5, SHA-1, or xxHash.
2. Key lookup
Hash the key → find position → walk clockwise → first node owns it. O(log N) with binary search.
3. Adding a node
New node only steals keys from its successor. Only ~K/N keys move. Everything else untouched.
4. Removing a node
Failed node's keys go to its clockwise neighbor. Only ~K/N keys move — the core advantage.
// Consistent hash ring — core implementation
class ConsistentHashRing {
  constructor(vnodes = 150) {
    this.ring   = new Map();  // position → nodeName
    this.sorted = [];          // sorted ring positions
    this.vnodes = vnodes;
  }
  addNode(node) {
    for (let i = 0; i < this.vnodes; i++) {
      const pos = hash(`${node}#${i}`);
      this.ring.set(pos, node);
      this.sorted.push(pos);
    }
    this.sorted.sort((a, b) => a - b);
  }
  getNode(key) {
    const pos = hash(key);
    let lo = 0, hi = this.sorted.length;
    while (lo < hi) {
      const mid = (lo + hi) >> 1;
      if (this.sorted[mid] < pos) lo = mid + 1;
      else hi = mid;
    }
    return this.ring.get(this.sorted[lo % this.sorted.length]);
  }
}
